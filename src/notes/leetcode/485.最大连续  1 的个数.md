# 485.最大连续  1 的个数

```ts
function findMaxConsecutiveOnes(nums: number[]): number {
  let tempCount = 0
  let maxCount = 0
  for (let i = 0; i < nums.length; i++) {
    if (nums[i] === 1) {
      tempCount++
      maxCount = Math.max(maxCount, tempCount)
    }
    else {
      tempCount = 0
    }
  }
  return maxCount
};
```
